redux-toolkit创建store

# RTK

  • redux-toolkit 简称 RTK

# RTK 作用

  • 用于简化 Redux 操作

# 安装

npm
npm install react-redux @reduxjs/toolkit -S
yarn
yarn add react-redux @reduxjs/toolkit

# 使用方式

  1. 在 store/index.js 去创建
import { configureStore, createSlice } from "@reduxjs/toolkit";
// 1.创建切片
const student = createSlice({
  // 用来设置action的type
  name: "stu",
  //   state初始值
  initialState: {
    sex: "男",
    name: "张三",
    age: 18,
    address: "武汉",
  },
  //   指定state的各种操作 action里有payload,type属性
  reducers: {
    setName(state, action) {
      state.name = action.payload.name;
    },
    setAge(state, action) {
      state.age = 28;
    },
  },
});
// 切片对象会自动帮助我们生成action
// {type:'stu/setName'} {type:'stu/setAge'}
export const { setName, setAge } = student.actions;

// 2.通过切片创建store
const store = configureStore({
  reducer: {
    stu: student.reducer,
  },
});
// 3.导出store
export default store;

  1. 在组件中使用，通过 useSelector 来获取指定 state，通过 useDispatch 来派发 action 方法
import { useDispatch, useSelector } from "react-redux";
import { setName } from "./store";

function App() {
  // useSelector() 用来加载state中的数据
  const student = useSelector((state) => state.stu);
  // 通过useDispatch() 来获取派发器对象
  const dispatch = useDispatch();
  // 获取构建器

  const eidtNameHandler = () => {
    // 比较麻烦，很容易出错
    // dispatch({ type: "stu/setName", paylod: "上网课" });
    // 推荐
    dispatch(setName({ name: "上网课" }));
  };
  return (
    <div className="App">
      <p>{student.name}</p>
      <p>{student.age}</p>
      <p>{student.sex}</p>
      <p>{student.address}</p>
      <button onClick={eidtNameHandler}>修改</button>
    </div>
  );
}

export default App;
