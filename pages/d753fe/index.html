# RTK Query

RTK 不仅帮助我们解决了 state 的问题，同时，它还为我们提供了 RTK Query 用来帮助我们处理数据加载问题。RTK Query 是一个强大的数据获取和缓存工具。在它的帮助下，web 应用中的加载变得十分简单，它使我们不再需要自己编写获取数据和缓存数据的逻辑。

Web 应用中加载数据时需要处理的问题：

  1. 根据不同的加载状态显示不同的 ui 组件
  2. 减少对相同数据重复发送请求
  3. 使用乐观更新，提示用户体验
  4. 在用户与 UI 交互时，管理缓存的生命周期

这些问题，RTKQ 都可以帮我们处理。

# 使用

RTKQ 已经继承在了 RTK 中，如果我们已经在项目中引入了 RTK 则不需再引起其余模块。如果你不想使用 RTKQ 给我们提供的方式请求的方式 (简单封装过的 fetch)，你还需要引入一下你要使用的方式请求的工具。

# 创建 Api 切片

RTKQ 中将一组相关功能统一封装到一个 Api 对象中。

# 示例 (基本使用)

  • 在 store 下创建 studentApi.js

    import { createApi, fetchBaseQuery } from "@reduxjs/toolkit/query/react";
// 创建Api对象
// createApi用来创建RTKQ中的API对象
// RTKQ的所有功能都需要通过该对象来进行
const studentApi = createApi({
  reducerPath: "studentApi", // API唯一标识
  baseQuery: fetchBaseQuery({ baseUrl: "http://localhost:1337/api" }), // 指定查询的基础信息，发送请求使用的工具
  endpoints(build, tagTypes) {
    return {
      // query查询
      getStudents: build.query({
        query: () => "/students",
      }),
      //   根据id进行查询
      getStudentById: build.query({
        query: (id) => "/students/" + id,
        keepUnusedDataFor: 0, // 设置缓存时间 单位s 设置0表示无缓存
      }),
    };
  }, // endpoints用来指定api各种功能 是一个方法，需要返回一个对象
});
export const { useGetStudentsQuery, useGetStudentByIdQuery } = studentApi;
export default studentApi;

  • 在 App 组件中使用

    import Students from "./components/Students/Students";
import { useGetStudentsQuery } from "./store/studentApi";

function App() {
  /* 
    refetch重新加载函数 刷新功能
    isFetching 加载状态 常用
    isLoading 数据第一次加载
    isSuccess 数据加载成功
    isUninitialized 请求是否还没有开始发送
    isError 是否有错误
    error：Error() 出错时存在
    currentData 当期参数的最新数据
    status 请求的状态
  */
  const { data, isLoading, isSuccess, refetch, isFetching, isUninitialized } =
    useGetStudentsQuery(null, {
      // 筛选
      // selectFromResult: (result) =>
      //   (result.data = result.data.filter((item) => item.attributes.age < 18)),
      pollingInterval: 0, // 设置轮训
      skip: false, // 是否跳过请求
      refetchOnMountOrArgChange: false, // 是否每次都重新加载数据 设置有效期和布尔值 true每次都重载数据
      refetchOnFocus: true, // 失去焦点重新获取 需要再store中设置开启支持 setupListeners(store.dispatch);
      refetchOnReconnect: false, //  重新连接时重新获取 需要再store中设置开启支持 setupListeners(store.dispatch);
    });
  console.log(data, "data");
  return (
    <div className="App">
      {isFetching && "数据加载中"}
      {isSuccess && <Students list={data.data}></Students>}
    </div>
  );
}

export default App;

# 增删改查

  1. 定义 api
import { createApi, fetchBaseQuery } from "@reduxjs/toolkit/query/react";
// 创建Api对象
// createApi用来创建RTKQ中的API对象
// RTKQ的所有功能都需要通过该对象来进行
const studentApi = createApi({
  reducerPath: "studentApi", // API唯一标识
  baseQuery: fetchBaseQuery({ baseUrl: "http://localhost:1337/api" }), // 指定查询的基础信息，发送请求使用的工具
  endpoints(build, tagTypes) {
    return {
      // query查询
      getStudents: build.query({
        query: () => "/students",
      }),
      //   根据id进行查询
      getStudentById: build.query({
        query: (id) => "/students/" + id,
        keepUnusedDataFor: 0, // 设置缓存时间 单位s 设置0表示无缓存
      }),
      addStudent: build.mutation({
        query: (stu) => ({
          method: "post",
          url: "/students",
          body: { data: stu },
        }),
      }),
      updateStudent: build.mutation({
        query: ({ id, ...putData }) => ({
          method: "put",
          url: "/students/" + id,
          body: { data: putData },
        }),
      }),
      delStudent: build.mutation({
        query: (id) => ({ method: "delete", url: "/students/" + id }),
      }),
    };
  }, // endpoints用来指定api各种功能 是一个方法，需要返回一个对象
});
export const {
  useGetStudentsQuery,
  useGetStudentByIdQuery,
  useAddStudentMutation,
  useUpdateStudentMutation,
  useDelStudentMutation,
} = studentApi;
export default studentApi;

# 数据标签

import { createApi, fetchBaseQuery } from "@reduxjs/toolkit/query/react";
// 创建Api对象
// createApi用来创建RTKQ中的API对象
// RTKQ的所有功能都需要通过该对象来进行
const studentApi = createApi({
  reducerPath: "studentApi", // API唯一标识
  baseQuery: fetchBaseQuery({ baseUrl: "http://localhost:1337/api" }), // 指定查询的基础信息，发送请求使用的工具
  endpoints(build, tagTypes) {
    return {
      // query查询
      getStudents: build.query({
        query: () => "/students",
        providesTags: [{ type: "student", id: "LIST" }],
      }),
      //   根据id进行查询
      getStudentById: build.query({
        query: (id) => "/students/" + id,
        // keepUnusedDataFor: 0, // 设置缓存时间 单位s 设置0表示无缓存
        providesTags: (result, error, id) => {
          return [{ type: "student", id }];
        },
      }),
      addStudent: build.mutation({
        query: (stu) => ({
          method: "post",
          url: "/students",
          body: { data: stu },
        }),
        invalidatesTags: [{ type: "student", id: "LIST" }],
      }),
      updateStudent: build.mutation({
        query: ({ id, ...putData }) => ({
          method: "put",
          url: "/students/" + id,
          body: { data: putData },
        }),
        invalidatesTags: (res, error, stu) => {
          return [
            { type: "student", id: stu.id },
            { type: "student", id: "LIST" },
          ];
        },
      }),
      delStudent: build.mutation({
        query: (id) => ({ method: "delete", url: "/students/" + id }),
        invalidatesTags: [{ type: "student", id: "LIST" }],
      }),
    };
  }, // endpoints用来指定api各种功能 是一个方法，需要返回一个对象
});
export const {
  useGetStudentsQuery,
  useGetStudentByIdQuery,
  useAddStudentMutation,
  useUpdateStudentMutation,
  useDelStudentMutation,
} = studentApi;
export default studentApi;
