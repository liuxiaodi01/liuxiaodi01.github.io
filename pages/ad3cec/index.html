vue实现md文件引入预览高亮

# 第一步

// npm install github-markdown-css 安装github样式
import 'github-markdown-css/github-markdown.css'

# 第二步在 vue.config.js

// npn i html-loader markdown-loader  // 解析md文件
// https://cli.vuejs.org/config/
module.exports = {
  // 会接收一个基于 webpack-chain 的 ChainableConfig 实例。
  // 允许对内部的 webpack 配置进行更细粒度的修改。
  chainWebpack: (config) => {
    // use show markdown file
    config.module
      .rule("md")
      .test(/\.md$/)
      .use("html-loader")
      .loader("html-loader")
      .end()
      .use("markdown-loader")
      .loader("markdown-loader")
      .end();
  },
};

# 第三步在 main.js

// 定义一个全局指令highlight 代码高亮的js
import hljs from "highlight.js";
// 引入样式 这里可能需要修改一下样式路径
import "highlight.js/scss/base16/atelier-cave-light.scss"; //样式文件
app.directive("highlight", (el) => {
  let blocks = el.querySelectorAll("pre code");
  blocks.forEach((block) => {
    hljs.highlightBlock(block);
  });
});

# 第四步在组件中使用

<!-- 使用指令v-highlight 高亮 -->
<template>
  <div class="markdown-body">
    <div v-html="docs" v-highlight ></div>
  </div>
</template>
<script>
import docs from "@/docs/installation.md";
export default {
  name: "Installation",
  components: {
    docs,
  },
  setup() {
  // vue2就在data里去定义赋值
    return {
      docs,
    };
  },
};
</script>
