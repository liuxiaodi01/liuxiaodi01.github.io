# jsonp 原理

# jsonp 主要用于 get 请求，需要与后端配合

<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <meta http-equiv="X-UA-Compatible" content="IE=edge" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <title>Document</title>
    <!-- jq对jsonp进行处理了的 实际上是script标签上传递query参数为callback=func -->
    <script src="https://cdn.bootcdn.net/ajax/libs/jquery/3.6.3/jquery.min.js"></script>
  </head>
  <body>
    <div id="app"></div>
    <script>
      window.onload = () => {
        $.ajax({
          dataType: "jsonp",
          method: "get",
          url: "http://localhost:9999/v1/list",
          success(data) {
            console.log(data, "sssssssss");

            document.getElementById("app").innerHTML = `<ul>${data.list.map(
              (v) => `<li>姓名:${v.name}-年龄:${v.age}-性别:${v.sex}</li>`
            )}<ul/>`;
          },
        });
      };
    </script>
  </body>
</html>

// express 开启端口
const express = require("express");
const app = express();
app.get("/v1/list", (req, res) => {
  const { callback } = req.query;
  const data = {
    code: 200,
    list: new Array(10)
      .fill(true)
      .map(() => ({ name: "张三", age: "李四", sex: "男" })),
  };
  res.send(`${callback}(${JSON.stringify(data)})`);
});
app.listen(9999, () => {
  console.log("http://localhost:9999");
});
